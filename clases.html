TDA: TDAArray

La clase TDAArray representa un tipo de dato abstracto para manejar arreglos.


class TDAArray:
    def __init__(self, size, value=None):
        self.__size = size
        self.__position = 0
        if size > 0:
            self.__array = []
            for i in range(0, self.__size):
                self.__array.append(None)
        else:
            self.__array = None
